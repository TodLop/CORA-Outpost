function createCoraSharedPoller({ channelName, intervalMs, fetchData, applyData, shouldPoll }) { const fallback = () => { const run = async () => { try { if (typeof shouldPoll === 'function' && !shouldPoll()) return; const payload = await fetchData(); applyData(payload); } catch (error) { console.error('Shared status poll failed:', error); } }; run(); return window.setInterval(run, intervalMs); }; if (!('BroadcastChannel' in window) || !('localStorage' in window)) { const fallbackTimer = fallback(); return { stop: () => window.clearInterval(fallbackTimer) }; } try { const probeKey = `${channelName}:probe`; window.localStorage.setItem(probeKey, '1'); window.localStorage.removeItem(probeKey); } catch (error) { const fallbackTimer = fallback(); return { stop: () => window.clearInterval(fallbackTimer) }; } const tabId = `${Date.now()}-${Math.random().toString(16).slice(2)}`; const channel = new BroadcastChannel(channelName); const leaderKey = `${channelName}:leader`; const leaderTtlMs = Math.max(intervalMs * 2, 8000); let pollTimer = null; let heartbeatTimer = null; const publish = (message) => channel.postMessage({ ...message, tabId, sentAt: Date.now() }); const readLeader = () => { try { return JSON.parse(window.localStorage.getItem(leaderKey) || 'null'); } catch (error) { return null; } }; const writeLeader = () => { const lease = { tabId, expiresAt: Date.now() + leaderTtlMs }; try { window.localStorage.setItem(leaderKey, JSON.stringify(lease)); const confirmed = readLeader(); if (confirmed?.tabId === tabId) { publish({ type: 'leader' }); return true; } } catch (error) { console.error('Shared status leader lock failed:', error); } return false; }; const isLeader = () => readLeader()?.tabId === tabId; const maybeAcquireLeader = () => { if (document.hidden) return false; const current = readLeader(); if (!current || current.expiresAt <= Date.now() || current.tabId === tabId) { return writeLeader(); } return false; }; const releaseLeader = () => { if (!isLeader()) return; try { window.localStorage.removeItem(leaderKey); } catch (error) { console.error('Shared status leader release failed:', error); } publish({ type: 'resign' }); }; const runLeaderPoll = async () => { if (!isLeader() && !maybeAcquireLeader()) return; if (typeof shouldPoll === 'function' && !shouldPoll()) return; try { const payload = await fetchData(); applyData(payload); publish({ type: 'data', payload }); } catch (error) { console.error('Shared status poll failed:', error); } }; channel.onmessage = (event) => { const message = event.data || {}; if (message.tabId === tabId) return; if (message.type === 'data') { applyData(message.payload); } else if (message.type === 'resign') { window.setTimeout(maybeAcquireLeader, 150 + Math.random() * 250); } }; document.addEventListener('visibilitychange', () => { if (document.hidden) { releaseLeader(); } else if (!document.hidden) { window.setTimeout(maybeAcquireLeader, 150 + Math.random() * 250); } }); window.addEventListener('storage', (event) => { if (event.key === leaderKey && !event.newValue) { window.setTimeout(maybeAcquireLeader, 150 + Math.random() * 250); } }); publish({ type: 'hello' }); pollTimer = window.setInterval(runLeaderPoll, intervalMs); heartbeatTimer = window.setInterval(() => { if (isLeader()) { writeLeader(); publish({ type: 'heartbeat' }); } else { maybeAcquireLeader(); } }, 2000); window.setTimeout(runLeaderPoll, 150 + Math.random() * 250); return { stop: () => { window.clearInterval(pollTimer); window.clearInterval(heartbeatTimer); releaseLeader(); channel.close(); } }; }
